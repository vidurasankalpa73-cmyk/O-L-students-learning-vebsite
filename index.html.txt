        const themeToggle = document.getElementById('themeToggle');

        // Theme Toggle
        let isDarkMode = false;
        themeToggle.addEventListener('click', () => {
            isDarkMode = !isDarkMode;
            document.body.setAttribute('data-theme', isDarkMode ? 'dark' : '');
            themeToggle.innerHTML = isDarkMode ? '<i class="fas fa-sun"></i>' : '<i class="fas fa-moon"></i>';
            
            // Save preference
            localStorage.setItem('theme', isDarkMode ? 'dark' : 'light');
        });

        // Load saved theme
        const savedTheme = localStorage.getItem('theme');
        if (savedTheme === 'dark') {
            isDarkMode = true;
            document.body.setAttribute('data-theme', 'dark');
            themeToggle.innerHTML = '<i class="fas fa-sun"></i>';
        }

        // Mobile Menu Toggle
        mobileMenu.addEventListener('click', () => {
            navLinks.classList.toggle('active');
            mobileMenu.innerHTML = navLinks.classList.contains('active') ? 
                '<i class="fas fa-times"></i>' : '<i class="fas fa-bars"></i>';
        });

        // Close mobile menu when clicking links
        document.querySelectorAll('.nav-links a').forEach(link => {
            link.addEventListener('click', () => {
                navLinks.classList.remove('active');
                mobileMenu.innerHTML = '<i class="fas fa-bars"></i>';
            });
        });

        // Navbar scroll effect
        window.addEventListener('scroll', () => {
            if (window.scrollY > 50) {
                navbar.style.boxShadow = '0 4px 20px rgba(0,0,0,0.1)';
            } else {
                navbar.style.boxShadow = '0 4px 6px -1px rgba(0, 0, 0, 0.1)';
            }
        });

        // Scroll Reveal Animation
        const revealObserver = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('active');
                }
            });
        }, { threshold: 0.1 });

        scrollElements.forEach(el => revealObserver.observe(el));

        // Subject Cards - View Notes
        document.querySelectorAll('.view-notes-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                e.stopPropagation();
                const card = btn.closest('.subject-card');
                const subject = card.dataset.subject;
                openNotesModal(subject);
            });
        });

        // Also make entire card clickable
        document.querySelectorAll('.subject-card').forEach(card => {
            card.addEventListener('click', (e) => {
                if (!e.target.classList.contains('btn-secondary')) {
                    const subject = card.dataset.subject;
                    openNotesModal(subject);
                }
            });
        });

        function openNotesModal(subject) {
            const data = subjectsData[subject];
            document.getElementById('modalTitle').textContent = data.title + ' Notes';
            
            // Generate lessons
            const lessonsContainer = document.getElementById('lessonsContainer');
            lessonsContainer.innerHTML = '';
            
            data.lessons.forEach((lesson, index) => {
                const lessonDiv = document.createElement('div');
                lessonDiv.className = 'lesson';
                lessonDiv.innerHTML = `
                    <div class="lesson-header" onclick="toggleLesson(${index})">
                        <span>${lesson.title}</span>
                        <i class="fas fa-chevron-down" id="icon-${index}"></i>
                    </div>
                    <div class="lesson-content" id="content-${index}">
                        <p>${lesson.content}</p>
                    </div>
                `;
                lessonsContainer.appendChild(lessonDiv);
            });

            // Generate quiz
            generateQuiz(data.quiz);
            
            notesModal.classList.add('active');
            document.body.style.overflow = 'hidden';
        }

        function toggleLesson(index) {
            const content = document.getElementById(`content-${index}`);
            const icon = document.getElementById(`icon-${index}`);
            
            content.classList.toggle('active');
            icon.style.transform = content.classList.contains('active') ? 'rotate(180deg)' : 'rotate(0deg)';
        }

        function generateQuiz(quizData) {
            const quizContent = document.getElementById('quizContent');
            const quizResult = document.getElementById('quizResult');
            quizContent.innerHTML = '';
            quizResult.innerHTML = '';
            
            let score = 0;
            let answered = 0;

            quizData.forEach((q, index) => {
                const questionDiv = document.createElement('div');
                questionDiv.className = 'quiz-question';
                questionDiv.innerHTML = `<p><strong>Q${index + 1}:</strong> ${q.question}</p>`;
                
                const optionsDiv = document.createElement('div');
                optionsDiv.className = 'quiz-options';
                
                q.options.forEach((opt, optIndex) => {
                    const optionBtn = document.createElement('div');
                    optionBtn.className = 'quiz-option';
                    optionBtn.textContent = opt;
                    optionBtn.addEventListener('click', () => {
                        if (optionBtn.classList.contains('correct') || optionBtn.classList.contains('wrong')) return;
                        
                        answered++;
                        if (optIndex === q.correct) {
                            optionBtn.classList.add('correct');
                            score++;
                        } else {
                            optionBtn.classList.add('wrong');
                            // Show correct answer
                            optionsDiv.children[q.correct].classList.add('correct');
                        }
                        
                        // Disable all options for this question
                        Array.from(optionsDiv.children).forEach(child => {
                            child.style.pointerEvents = 'none';
                        });

                        if (answered === quizData.length) {
                            quizResult.innerHTML = `
                                <i class="fas fa-trophy"></i> 
                                You scored ${score}/${quizData.length}! 
                                ${score === quizData.length ? 'Perfect! 🎉' : 'Good effort! Keep studying!'}
                            `;
                            quizResult.style.color = score === quizData.length ? '#22c55e' : '#f59e0b';
                        }
                    });
                    optionsDiv.appendChild(optionBtn);
                });
                
                questionDiv.appendChild(optionsDiv);
                quizContent.appendChild(questionDiv);
            });
        }

        // Close Modal
        closeModal.addEventListener('click', () => {
            notesModal.classList.remove('active');
            document.body.style.overflow = 'auto';
        });

        notesModal.addEventListener('click', (e) => {
            if (e.target === notesModal) {
                notesModal.classList.remove('active');
                document.body.style.overflow = 'auto';
            }
        });

        // Contact Form Validation
        contactForm.addEventListener('submit', (e) => {
            e.preventDefault();
            
            const name = document.getElementById('name');
            const email = document.getElementById('email');
            const message = document.getElementById('message');
            const submitBtn = document.getElementById('submitBtn');
            
            let isValid = true;

            // Name validation
            if (name.value.trim() === '') {
                document.getElementById('nameError').classList.add('show');
                isValid = false;
            } else {
                document.getElementById('nameError').classList.remove('show');
            }

            // Email validation
            const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
            if (!emailRegex.test(email.value)) {
                document.getElementById('emailError').classList.add('show');
                isValid = false;
            } else {
                document.getElementById('emailError').classList.remove('show');
            }

            // Message validation
            if (message.value.trim().length < 10) {
                document.getElementById('messageError').classList.add('show');
                isValid = false;
            } else {
                document.getElementById('messageError').classList.remove('show');
            }

            if (isValid) {
                // Simulate form submission
                submitBtn.disabled = true;
                submitBtn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Sending...';
                
                setTimeout(() => {
                    document.getElementById('successMessage').classList.add('show');
                    contactForm.reset();
                    submitBtn.disabled = false;
                    submitBtn.innerHTML = 'Send Message <i class="fas fa-paper-plane"></i>';
                    
                    setTimeout(() => {
                        document.getElementById('successMessage').classList.remove('show');
                    }, 5000);
                }, 1500);
            }
        });

        // Real-time validation removal
        ['name', 'email', 'message'].forEach(fieldId => {
            document.getElementById(fieldId).addEventListener('input', () => {
                document.getElementById(fieldId + 'Error').classList.remove('show');
            });
        });

        // Keyboard accessibility for modal
        document.addEventListener('keydown', (e) => {
            if (e.key === 'Escape' && notesModal.classList.contains('active')) {
                notesModal.classList.remove('active');
                document.body.style.overflow = 'auto';
            }
        });
